[docs]class DictIterator(object):
    def __init__(self, dictionary, stopValue):
        self.dictionary = dictionary
        self.stopValue = stopValue
        self.count = 0
    def __next__(self):
        dictionary = self.dictionary
        if dictionary:
            key, value = dictionary.popitem()
            if not value == self.stopValue:
                self.count += 1
                return key
        raise StopIteration()
    def __iter__(self):
        self.count = 0
        return self
